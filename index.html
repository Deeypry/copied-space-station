Атмосферный контроль

ИСХОДНЫЙ КОД

Прототип игры написан на Python с использованием Pygame. Ниже представлена ключевая часть кода — система атмосферного контроля.


class AtmosGrid:
    """Отсек станции с атмосферой"""
    
    def __init__(self, name: str, is_breached: bool = False):
        self.name = name
        self.gases: Dict[GasType, Gas] = {}
        self.is_breached = is_breached
        
        # Инициализируем стандартную атмосферу
        self.gases[GasType.OXYGEN] = Gas(GasType.OXYGEN, 800.0, 293.0)
        self.gases[GasType.NITROGEN] = Gas(GasType.NITROGEN, 3200.0, 293.0)
    
    @property
    def pressure(self) -> float:
        """Вычисляем давление по уравнению идеального газа"""
        if self.total_moles == 0:
            return 0.0
        return (self.total_moles * self.IDEAL_GAS_CONSTANT * 
                self.average_temperature) / self.GRID_VOLUME
    
    def update(self):
        """Обновление состояния атмосферы каждый тик"""
        # 1. Утечка в космос при повреждении
        if self.is_breached:
            leak_factor = 0.05
            for gas_type, gas in list(self.gases.items()):
                gas.amount -= gas.amount * leak_factor
        
        # 2. Система жизнеобеспечения добавляет кислород
        if self.pressure < 95.0:
            if GasType.OXYGEN in self.gases:
                self.gases[GasType.OXYGEN].amount += 50.0
            else:
                self.gases[GasType.OXYGEN] = Gas(GasType.OXYGEN, 50.0, 293.0)
